manuRuvAcha |
    namo namo vareNaya varadAyA'aMshumAline |
    jyotirMaya namastubhyamAmanatAyAjitAte || 1||

trilokachakShute tubhya trigunAyAmRutAya cha |
    namo dharmAya haMsAya jagajananahetave || 2||

naranArIshararIrAya namo meDustamAyAte |
    prajnAnAyA-khileAshAya saptAshvAya trimUrtaye || 3||

namo vyAhRitirUpAya trilakShAyA'haMshAgAmine |
    hariAshvAya namastubhyamo hari-tavAhave || 4||

eka-lakShavilakShAya bahulakShAya daNNe |
    ekasaMsthadvisaMsthAya bahusaMsthAya te namaH || 5||

shaktitrayAya shuklAya ravaye parameShThine |
    tvaM shivastvaM hariRdeva tvaM brahmA tvaM divaspatiH || 6||

tvAM OmkAro vashT-kAraH svadhA svAhA tvameva hi |
    tvAmRute paramAtmA na tA-paShyAmi daivatamaH || 7||
