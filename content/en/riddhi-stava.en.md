shrimanvrishabhailesa vardhataam vijee bhavaan |
divyam tvadeeyamaishvaryam nirmaaryaadam vijmbhataam || 1 ||

deveebhausha ayudhairnityairmuktairmokshaikalakshanaih |
sattvottaraistvadeeyaisha sangah staatarasastava || 2 ||

praarakagopuravapraasaadamanimantapaah |
shalimudgatiladeenaam shaalah shailakulojjvalaah || 3 ||

ratnakaancanakausheya kshoumakramukashaalikaah |
shayyaagruhaani paryankavaryaah sthuulaasanani ca || 4 ||

kanatkanakabhrngaara patadgrahakalaacikaa |
chatraacaaramukhyaashca santu nityaah paricchadaah || 5 ||

astu nistulamavyagran nityamabhyarchanam tava |
pakse pakse vivardhantaam maasimaaasi mahotsavaah || 6 ||

manikaancanitraani bhaushanaanyambaraani ca |
kaashmirasaarakastooke rkarpuuraadyanulepanam || 7 ||

komalaani ca daamaani kusamih saurabhautkaraaih |
dhuupaah karpuuradeepaashca santu santatameva te || 8 ||

nattageetayutam vaadyam nityamatra vivardhataam |
shrotreshu sudhaadhaaraah kalpantaam kaahaleesvanaah || 9 ||

kandamoolaphalodagam kaalekaale catuavidham |
suupaapuupaghrtaksheeara arkaasahitam havih || 10 ||

ghanasaearashilodagraih kramukaash tadalaaih saha |
vimalaani ca taambhoole edalaani sveekuru prabho || 11 ||

preetibhitiyuto bhu yaadbhu yaan parijanastava |
bhaktimanto bhajantutvaam puraa jaanapadaastathaa || 12 ||

dharaaneedhararatnaani vitaramantu ciram tava |
kaikaryamakhilam sarve kurvantu kshonipaalaakah || 13 ||

premadigdadrishah svairam prekshamaanaastvadaan anam |
mahaantah santatam santo mangalaani prayunjataam || 14 ||

evameva bhavenityam palayana kusalee bhavaan |
maamaahirama shrmaan vardhataamabhivardhataam || 15 ||

patyuh pratyahamittham yah praarthayeta samucchrayam |
prasaadasumukhah shrmaan pashyatyenaam parah pumaan || 16 ||
